## Getting Started with Examples

### Prerequisites

<Steps>
<Step title="Clone Repository">
  ```bash
  git clone https://github.com/odin/odin-sdk.git
  cd odin-sdk
  ```
</Step>

<Step title="Install Dependencies">```bash npm install ```</Step>

<Step title="Build SDK">```bash npm run build ```</Step>

<Step title="Run Examples">
  ```bash

  npx tsx examples/auth-example.ts
  npx tsx examples/token-example.ts
  npx tsx examples/user-example.ts
  ```

## Integration Examples

### React Integration

Example React component using the SDK:

```typescript react-example.tsx
import React, { useState } from 'react';
import { OdinSDK } from '@odin/sdk';

export function OdinAuthButton() {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [user, setUser] = useState<string | null>(null);

  const odin = new OdinSDK({
    baseUrl: 'https://api.odin.fun/dev',
  });

  const handleAuthentication = async () => {
    try {
      // Get address from wallet
      const address = await window.unisat.getAddress();

      // Prepare authentication
      const prepareResult = await odin.auth.prepare(address);

      // Sign message
      const signature = await window.unisat.signMessage(prepareResult.message);
      const publicKey = await window.unisat.getPublicKey();

      // Complete authentication
      const authResult = await odin.auth.login({
        address,
        message: prepareResult.message,
        signature,
        publicKey,
        signatureType: 'Bip322Simple',
      });

      setIsAuthenticated(true);
      setUser(authResult.principalId);
    } catch (error) {
      console.error('Authentication failed:', error);
    }
  };

  return (
    <div>
      {!isAuthenticated ? (
        <button onClick={handleAuthentication}>
          Sign In with Bitcoin
        </button>
      ) : (
        <div>
          <p>Welcome! Principal: {user}</p>
          <button onClick={() => odin.auth.signOut()}>
            Sign Out
          </button>
        </div>
      )}
    </div>
  );
}
```

### Next.js Integration

Example Next.js API route:

```typescript api-route.ts
// pages/api/odin-data.ts
import { NextApiRequest, NextApiResponse } from 'next';
import { OdinSDK } from '@odin/sdk';

export default async function handler(req: NextApiRequest, res: NextApiResponse) {
  const odin = new OdinSDK({
    baseUrl: process.env.ODIN_API_BASE_URL || 'https://api.odin.fun/v1',
  });

  try {
    switch (req.method) {
      case 'GET':
        const tokens = await odin.tokens.getTokens({
          trading: true,
          limit: 10,
        });
        res.status(200).json(tokens);
        break;

      default:
        res.setHeader('Allow', ['GET']);
        res.status(405).end(`Method ${req.method} Not Allowed`);
    }
  } catch (error) {
    console.error('API error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
}
```

### Debugging Examples

Enable comprehensive logging for debugging:

```typescript debug-config.ts
const odin = new OdinSDK({
  baseUrl: 'https://api.odin.fun/dev',
  enableLogging: true, // Enable SDK logging
  timeout: 60000, // Increase timeout for debugging
});

// Enable browser console logging
console.log = (...args) => {
  console.info(`[${new Date().toISOString()}]`, ...args);
};
```
